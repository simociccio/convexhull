48  int first = 0, last =v.size()-1, mid;
49  while (first <= last){
50  mid = (first + last) / 2;

116  if(start->info().nesting_level != -1){
117  return;
118  }
119  std::list<CDT::Face_handle> queue;
120  queue.push_back(start);
121  while(! queue.empty()){
122  CDT::Face_handle fh = queue.front();
123  queue.pop_front();
124  if(fh->info().nesting_level == -1){
125  fh->info().nesting_level = index;
126  for(int i = 0; i < 3; i++){
127  CDT::Edge e(fh,i);

131  else queue.push_back(n);

155 inline void getRotationMatrix(Vec3 axis, double angle, double m[][3]) {
156  axis.normalize();
157  double cosa = cos(angle);
158  double sina = sin(angle);
159  m[0][0] = cosa + (axis.x() * axis.x())*(1-cosa);
160  m[0][1] = axis.x() * axis.y() * (1-cosa) - axis.z() * sina;
161  m[0][2] = axis.x() * axis.z() * (1-cosa) + axis.y() * sina;
162  m[1][0] = axis.y() * axis.x() * (1-cosa) + axis.z() * sina;
163  m[1][1] = cosa + (axis.y() * axis.y())*(1-cosa);
164  m[1][2] = axis.y() * axis.z() * (1-cosa) - axis.x() * sina;
165  m[2][0] = axis.z() * axis.x() * (1-cosa) - axis.y() * sina;
166  m[2][1] = axis.z() * axis.y() * (1-cosa) + axis.x() * sina;
167  m[2][2] = cosa + (axis.z() * axis.z())*(1-cosa);
